import { cookies } from "next/headers";
import { NextRequest, NextResponse } from "next/server";

export async function middleware(request: NextRequest) {
    const path = request.nextUrl.pathname;
    const token = cookies().get('token')?.value;

    // const isPublicPath = path === '/login' || path === '/register' || path === '/';

    //! If there's no token and the path is not public, redirect to login
    if (!token
        // && !isPublicPath
    ) {
        return NextResponse.redirect(new URL('/login', request.nextUrl));
    }

    //? If there's no token and the path is public, redirect to /browseCourses
    if (!token
        // && isPublicPath
    ) {
        return NextResponse.redirect(new URL(`/browseCourses`, request.nextUrl));
    }

    //* If there's a token and the path is not-public, redirect to /:uid/browseCourses
    if (token
        // && !isPublicPath
    ) {
        const uid = getTokenUid(token);
        return NextResponse.redirect(new URL(`/${uid}/browseCourses`, request.nextUrl));
    }

    //! --- ONly for testing after testing update this ---
    if (token && path.includes('/instructor')) {
        return NextResponse.redirect(new URL(`${path}`, request.nextUrl));
    }

    if (token && path.includes('/profile')) {
        return NextResponse.redirect(new URL(`${path}`, request.nextUrl));
    }
}

async function getTokenUid(token: string): Promise<string | null> {
    try {
        const response = await fetch('https://localhost:3000/api/auth/me', {
            headers: {
                Authorization: `Bearer ${token}`,
            },
        });

        if (!response.ok) {
            // Handle non-OK responses, e.g., unauthorized
            console.error(`Failed to fetch user data. Status: ${response.status}`);
            return null;
        }

        const data = await response.json();
        // Assuming the response has a user ID property, modify this based on your actual response structure
        const userId = data.userId;
        return userId;
    } catch (error) {
        console.error('Error fetching user data:', error);
        return null;
    }
}

export const config = {
    matcher: [
        '/',
        '/register',
        '/login',
        '/profile/:path*',
        '/instructor/:path*',
        // '/((?!.+\\.[\\w]+$|_next).*)', '/', '/(api|trpc)(.*)'
    ],
}
