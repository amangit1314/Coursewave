import Redis from "ioredis";
import { promisify } from "util";

// Create Redis client with error handling
const createRedisClient = () => {
  try {
    const client = new Redis({
      host: process.env.REDIS_HOST || "localhost",
      port: parseInt(process.env.REDIS_PORT || "6379"),
      // password: process.env.REDIS_PASSWORD || undefined,
      // Only include password if set
      ...(process.env.REDIS_PASSWORD && {
        password: process.env.REDIS_PASSWORD,
      }),
      db: parseInt(process.env.REDIS_DB || "0"),
      retryStrategy: (times) => {
        const delay = Math.min(times * 50, 2000);
        return delay;
      },
    });

    client.on("error", (err) => {
      console.error("Redis connection error:", err);
    });

    client.on("connect", () => {
      console.log("Redis connected successfully");
    });

    return client;
  } catch (error) {
    console.error("Failed to create Redis client:", error);
    // Return a mock client that logs operations but doesn't actually do anything
    return {
      get: async (key: string) => {
        console.warn(`Redis get called with key: ${key} (Redis not available)`);
        return null;
      },
      set: async (key: string, value: string) => {
        console.warn(`Redis set called with key: ${key} (Redis not available)`);
        return "OK";
      },
      setEx: async (key: string, seconds: number, value: string) => {
        console.warn(
          `Redis setEx called with key: ${key}, seconds: ${seconds} (Redis not available)`
        );
        return "OK";
      },
      del: async (key: string) => {
        console.warn(`Redis del called with key: ${key} (Redis not available)`);
        return 0;
      },
      keys: async (pattern: string) => {
        console.warn(
          `Redis keys called with pattern: ${pattern} (Redis not available)`
        );
        return [];
      },
    } as any;
  }
};

export const redisClient = createRedisClient();
export const rateLimitRedis = createRedisClient(); // Add separate client for rate limiting

// Cache middleware
export const cacheMiddleware = (duration: number) => {
  return async (req: any, res: any, next: any) => {
    // Skip caching if Redis is not available
    if (!redisClient || typeof redisClient.get !== "function") {
      return next();
    }

    const key = `cache:${req.originalUrl}`;

    try {
      const cachedResponse = await redisClient.get(key);

      if (cachedResponse) {
        return res.json(JSON.parse(cachedResponse));
      }

      // Store original send function
      const originalSend = res.send;

      // Override send function to cache the response
      res.send = function (body: any) {
        // Restore original send
        res.send = originalSend;

        // Cache the response
        if (res.statusCode === 200) {
          redisClient
            .setEx(key, duration, JSON.stringify(body))
            .catch((err: Error) =>
              console.error("Error caching response:", err)
            );
        }

        // Send the response
        return originalSend.call(this, body);
      };

      next();
    } catch (error) {
      console.error("Cache middleware error:", error);
      next();
    }
  };
};

// Clear cache for a specific key pattern
export const clearCache = async (pattern: string) => {
  try {
    const keys = await redisClient.keys(`cache:${pattern}`);
    if (keys.length > 0) {
      await redisClient.del(keys);
      console.log(
        `Cleared ${keys.length} cache entries for pattern: ${pattern}`
      );
    }
  } catch (error) {
    console.error("Error clearing cache:", error);
  }
};
